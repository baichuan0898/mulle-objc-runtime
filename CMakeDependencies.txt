   include( ${DEPENDENCIES_DIR}/include/mulle_concurrent/CMakeDependencies.txt)

         ${MULLE_OBJC_RUNTIME_LIBRARY}

      )
   endif()

      set( OBJC_STARTUP_LIBRARY
         ${MULLE_OBJC_RUNTIME_STARTUP_LIBRARY}

      )
   endif()
