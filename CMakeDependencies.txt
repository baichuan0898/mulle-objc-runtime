
#
# The ifs are there so, that if someone has run find_libary already
# it doesn't add duplicates to DEPENDENCY_LIBRARIES
#

if( NOT MULLE_ABA_LIBRARY)
   find_library( MULLE_ABA_LIBRARY NAMES mulle_aba)
   message(STATUS "MULLE_ABA_LIBRARY is ${MULLE_ABA_LIBRARY}")
   set( C_DEPENDENCY_LIBRARIES
${C_DEPENDENCY_LIBRARIES}
${MULLE_ABA_LIBRARY}
CACHE INTERNAL "need to cache this"
)
endif()
